# Copy all scripts to the build directory, i.e. the directory in which we

#
set(SIMUL_SCRIPTS              vis.mac)

foreach(_script ${SIMUL_SCRIPTS})
  configure_file(
    ${PROJECT_SOURCE_DIR}/simul/${_script}
    ${PROJECT_BINARY_DIR}/simul/${_script}
    COPYONLY
    )
